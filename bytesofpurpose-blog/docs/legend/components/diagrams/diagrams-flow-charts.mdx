# Flow Chart
<div id="flowchart_diagram">
  <script>
    var code = `
      st=>start: Start
      op=>operation: Perform Operation
      cond=>condition: Is condition met?
      e=>end: End

      st->op->cond
      cond(yes)->e
      cond(no)->op
    `;

    var diagram = flowchart.parse(code);
    diagram.drawSVG('diagram');
  </script>
</div>
